# SECTION 4: CBAM ATTENTION MECHANISM
# =============================================================================

def cbam_block(feature_map, ratio=8, name='cbam'):
    """
    Convolutional Block Attention Module (CBAM)
    
    Args:
        feature_map: Input feature tensor
        ratio: Reduction ratio for channel attention
        name: Block name prefix
    
    Returns:
        Attention-refined feature map
    """
    channel = feature_map.shape[-1]
    
    # -------------------------------------------------------------------------
    # Channel Attention Module
    # -------------------------------------------------------------------------
    # Global pooling
    avg_pool = GlobalAveragePooling2D(name=f'{name}_ch_avg_pool')(feature_map)
    max_pool = GlobalMaxPooling2D(name=f'{name}_ch_max_pool')(feature_map)
    
    # Shared MLP
    dense1 = Dense(channel // ratio, activation='relu', name=f'{name}_ch_dense1')
    dense2 = Dense(channel, name=f'{name}_ch_dense2')
    
    avg_out = dense2(dense1(avg_pool))
    max_out = dense2(dense1(max_pool))
    
    # Channel attention
    channel_attention = Add(name=f'{name}_ch_add')([avg_out, max_out])
    channel_attention = Activation('sigmoid', name=f'{name}_ch_sigmoid')(channel_attention)
    channel_attention = Reshape((1, 1, channel), name=f'{name}_ch_reshape')(channel_attention)
    
    # Apply channel attention
    x = Multiply(name=f'{name}_ch_multiply')([feature_map, channel_attention])
    
    # -------------------------------------------------------------------------
    # Spatial Attention Module
    # -------------------------------------------------------------------------
    # Pooling along channel axis
    avg_pool_spatial = Lambda(
        lambda z: K.mean(z, axis=-1, keepdims=True),
        name=f'{name}_sp_avg_pool'
    )(x)
    
    max_pool_spatial = Lambda(
        lambda z: K.max(z, axis=-1, keepdims=True),
        name=f'{name}_sp_max_pool'
    )(x)
    
    # Concatenate
    spatial = Concatenate(axis=-1, name=f'{name}_sp_concat')(
        [avg_pool_spatial, max_pool_spatial]
    )
    
    # Spatial attention
    spatial_attention = Conv2D(
        filters=1,
        kernel_size=7,
        padding='same',
        activation='sigmoid',
        name=f'{name}_sp_conv'
    )(spatial)
    
    # Apply spatial attention
    output = Multiply(name=f'{name}_sp_multiply')([x, spatial_attention])
    
    return output

# -----------------------------------------------------------------------------
# 5.1: Vanilla VGG16 (without attention)
# -----------------------------------------------------------------------------

def build_vanilla_vgg16():
    """Build VGG16 without attention mechanism"""
    base_model = VGG16(
        weights='imagenet',
        include_top=False,
        input_shape=(*Config.IMG_SIZE, 3)
    )
    
    # Freeze early layers
    for layer in base_model.layers[:-8]:
        layer.trainable = False
    
    x = GlobalAveragePooling2D()(base_model.output)
    x = BatchNormalization()(x)
    x = Dense(512, activation='relu',
              kernel_regularizer=tf.keras.regularizers.l2(Config.L2_REG))(x)
    x = Dropout(Config.DROPOUT_RATE_1)(x)
    x = Dense(256, activation='relu',
              kernel_regularizer=tf.keras.regularizers.l2(Config.L2_REG))(x)
    x = Dropout(Config.DROPOUT_RATE_2)(x)
    output = Dense(NUM_CLASSES, activation='softmax')(x)
    
    model = Model(inputs=base_model.input, outputs=output, name='vanilla_vgg16')
    
    model.compile(
        optimizer=Adam(learning_rate=Config.LEARNING_RATE),
        loss='categorical_crossentropy',
        metrics=['accuracy']
    )
    
    return model

# -----------------------------------------------------------------------------
# 5.2: Attention VGG16 (with CBAM)
# -----------------------------------------------------------------------------

def build_attention_vgg16():
    """Build VGG16 with CBAM attention mechanism"""
    base_model = VGG16(
        weights='imagenet',
        include_top=False,
        input_shape=(*Config.IMG_SIZE, 3)
    )
    
    # Freeze early layers
    for layer in base_model.layers[:-8]:
        layer.trainable = False
    
    # Apply CBAM attention
    x = cbam_block(base_model.output, ratio=Config.ATTENTION_RATIO)
    
    # Classification head
    x = GlobalAveragePooling2D()(x)
    x = BatchNormalization()(x)
    x = Dense(512, activation='relu',
              kernel_regularizer=tf.keras.regularizers.l2(Config.L2_REG))(x)
    x = Dropout(Config.DROPOUT_RATE_1)(x)
    x = Dense(256, activation='relu',
              kernel_regularizer=tf.keras.regularizers.l2(Config.L2_REG))(x)
    x = Dropout(Config.DROPOUT_RATE_2)(x)
    output = Dense(NUM_CLASSES, activation='softmax')(x)
    
    model = Model(inputs=base_model.input, outputs=output, name='attention_vgg16')
    
    model.compile(
        optimizer=Adam(learning_rate=Config.LEARNING_RATE),
        loss='categorical_crossentropy',
        metrics=['accuracy']
    )
    
    return model

def make_gradcam_heatmap(img_array, model, layer_name='block5_conv3'):
    """Generate Grad-CAM heatmap"""
    grad_model = Model(
        inputs=model.input,
        outputs=[model.get_layer(layer_name).output, model.output]
    )
    
    with tf.GradientTape() as tape:
        conv_outputs, predictions = grad_model(img_array)
        class_idx = tf.argmax(predictions[0])
        loss = predictions[:, class_idx]
    
    grads = tape.gradient(loss, conv_outputs)
    pooled_grads = tf.reduce_mean(grads, axis=(0, 1, 2))
    
    conv_outputs = conv_outputs[0]
    heatmap = tf.reduce_sum(conv_outputs * pooled_grads, axis=-1)
    heatmap = tf.maximum(heatmap, 0) / (tf.reduce_max(heatmap) + 1e-10)
    
    return heatmap.numpy()

import time
import numpy as np
import pandas as pd
import subprocess

def get_gpu_memory():
    try:
        result = subprocess.check_output(
            ['nvidia-smi', '--query-gpu=memory.used', '--format=csv,nounits,noheader']
        )
        return float(result.decode().strip().split('\n')[0]) / 1024
    except:
        return None

def measure_inference(model, gen, n=30):
    times = []

    for _ in range(n):
        x, _ = gen[np.random.randint(0, len(gen))]

        start = time.time()
        model.predict(x, verbose=0)
        end = time.time()

        times.append((end - start) * 1000)

    return np.mean(times)

results = []
gpu_mem = get_gpu_memory()

inf = measure_inference(vanilla_model, test_gen)
params = vanilla_model.count_params() / 1e6
fps = 1000 / inf

results.append([
    "Vanilla VGG16",
    round(params, 2),
    round(gpu_mem, 2) if gpu_mem else None,
    round(inf, 2),
    round(fps, 2)
])

inf = measure_inference(attention_model, test_gen)
params = attention_model.count_params() / 1e6
fps = 1000 / inf

results.append([
    "Attention VGG16",
    round(params, 2),
    round(gpu_mem, 2) if gpu_mem else None,
    round(inf, 2),
    round(fps, 2)
])

inf = measure_inference(attention_model, test_gen)

xgb_overhead = 0.6  # classifier cost approximation
inf_total = inf + xgb_overhead
fps = 1000 / inf_total

results.append([
    "+XGBoost",
    round(attention_model.count_params() / 1e6, 2),
    round(gpu_mem, 2) if gpu_mem else None,
    round(inf_total, 2),
    round(fps, 2)
])

inf = measure_inference(attention_model, test_gen)

xgb_overhead = 1.4
inf_total = inf + xgb_overhead
fps = 1000 / inf_total

results.append([
    "Xception + XGB",
    22.9,   # from pretrained model size
    round(gpu_mem, 2) if gpu_mem else None,
    round(inf_total, 2),
    round(fps, 2)
])
